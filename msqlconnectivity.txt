def insert(id,name, age, city):
    res = con.cursor()
    sql = "insert into users (id,name,age,city) values (%s,%s,%s,%s)"
    user = (id,name, age, city)
    res.execute(sql, user)
    con.commit()
    print("Data Insert Success")
